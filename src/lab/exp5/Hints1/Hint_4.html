#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
int isAlphabet(char x){
	return (x>='a' && x<='z')||(x>='A' && x<='Z');
}
int main(){
char s[1001];
scanf("%[^\n]s",s);
int i=0,j=strlen(s)-1;
int flag=1;
while(i < j){
	
	while(i < j && !isAlphabet(s[i]))i++;	
	while(i < j && !isAlphabet(s[j]))j--;	
	if(i>=j)
		break;
	if( (s[i]==s[j]||abs(s[i]-s[j])==32)) 
		i++,j--;
	else{
		flag=0;
		break;
	}
}
if(flag)
	printf("YES\n");
else
	 printf("NO\n");

	return 0;
}
